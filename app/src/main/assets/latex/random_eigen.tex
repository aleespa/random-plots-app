
\text{For } \theta_i \sim \mathcal{U}[0, 2\pi] \text{ with } i = 1, 2, \ldots, 3n, \\
\text{complex numbers } z_i = e^{i\theta_i} \text{ are sampled.} \\
\text{These values are reshaped into } n \text{ sets of three pairs: } \\
t = \{z_1, z_2, z_3\}, z_i \in \mathbb{C} \\

\text{A random matrix } M(t, r_1, r_2, r_3) \text{ is generated as follows: } \\
M(t, r_1, r_2, r_3) = \begin{bmatrix}
i & -r_3 & -t_3 & r_1 & i \\
-1 & r_3 & 0 & 1 & 1 \\
t_2 & r_2 & -i & t_3 & i \\
i & t_1 & i & i & i \\
i & 2 & -1 & -1 & i
\end{bmatrix} \\

\text{The eigenvalues } \lambda_i \text{ of each matrix } M \text{ are computed, } \\
\lambda_i \in \mathbb{C}, \text{ yielding complex values for } x = \Re(\lambda_i) \text{ and } y = \Im(\lambda_i). \\

\text{Total number of points plotted: } n \times 5 = 50,000 \text{ (for } n = 10,000\text{)}.
